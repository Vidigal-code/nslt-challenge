REMOVE ALL DOCKER WINDOWS


:: Stop all containers
for /f "tokens=*" %i in ('docker ps -aq') docker stop %i

:: Remove all containers
for /f "tokens=*" %i in ('docker ps -aq') docker rm %i

:: Remove all images
for /f "tokens=*" %i in ('docker images -q') docker rmi -f %i

:: Remove all volumes
for /f "tokens=*" %i in ('docker volume ls -q') docker volume rm %i

:: Remove all networks
for /f "tokens=*" %i in ('docker network ls -q') do docker network rm %i

:: Remove builder cache and old builds
docker builder prune -a -f
